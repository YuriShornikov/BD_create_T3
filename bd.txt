CREATE TABLE IF NOT EXISTS musical_genres 
(
	musical_genre_id INT PRIMARY KEY,
	name VARCHAR(60) NOT NULL
)

CREATE TABLE IF NOT EXISTS musicians
(
	musician_id INT PRIMARY KEY,
	name VARCHAR(60) NOT NULL
)

CREATE TABLE IF NOT EXISTS musical_genres_musicias
(
	musical_genres_musicias_id INT PRIMARY KEY,
	musician_id INT NOT NULL references musicians(musician_id),
	musical_genre_id INT NOT NULL references musical_genres(musical_genre_id)
)

CREATE TABLE IF NOT EXISTS albums
(
	album_id INT PRIMARY KEY,
	name VARCHAR(60) NOT NULL,
	album_release INT NOT NULL 
)

CREATE TABLE IF NOT EXISTS musicians_albums
(
	musicians_albums_id INT PRIMARY KEY,
	musician_id INT NOT NULL references musicians(musician_id),
	album_id INT NOT NULL references albums(album_id)
)

CREATE TABLE IF NOT EXISTS digests
(
	digests_id INT PRIMARY KEY,
	name VARCHAR(60) NOT NULL,
	digests_release INT NOT NULL 
)

CREATE TABLE IF NOT EXISTS tracks
(
	track_id INT PRIMARY KEY,
	name VARCHAR(60) NOT NULL,
	time INT NOT null,
	album_id INT NOT NULL references albums(album_id),
	digests_id INT NOT NULL references digests(digests_id),
)
